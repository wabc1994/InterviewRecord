    int low = 0;
    int high = array.length;
    while (low < high) {
        int mid = low + (high - low) / 2;
        if (array[mid] == key) {
            return mid;
        }
 
        if (array[mid] < key) {
            low = mid + 1;
        } else {
            high = mid;
        }
    }
    return -1;
}
}
